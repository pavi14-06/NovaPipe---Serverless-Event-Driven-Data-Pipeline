variable "aws_region" {
  description = "AWS region to deploy into"
  type        = string
  default     = "us-east-1"
}

variable "pipeline_env" {
  description = "Pipeline environment (production, staging, development)"
  type        = string
  default     = "production"

  validation {
    condition     = contains(["production", "staging", "development"], var.pipeline_env)
    error_message = "pipeline_env must be one of: production, staging, development"
  }
}
